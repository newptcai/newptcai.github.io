Tiling chessboard with L-shaped tiles¶

This is a demonstration that a $2^n \times 2^n$ chessboard with one tile removed can always be coverd by L-shaped tiles.

import matplotlib.pyplot as plt
import numpy as np
from ipywidgets import interact, interactive, fixed

board_cmap = plt.cm.get_cmap('RdYlGn')
board_cmap.set_under('white')
def draw_board(board):
    fig, ax = plt.subplots(figsize=(10,10))
    size = np.size(board, axis=0)
    plt.title(f"Chess board of size {size}x{size}")
    ax.matshow(board, cmap=board_cmap, vmin=.001);

def tile_board(board):
    size = np.size(board, axis=0)
    color = np.random.random()
    if size == 2:
        for (x, y), value in np.ndenumerate(board):
            if value != 0:
                board[x, y] = color
    else:
        half = int(size/2)
        slice_size = half**2
        shift = np.array([[x,y] for x in range(2) for y in range(2)], np.int32, )
        top_left = shift * half
        center = np.tile([half-1, half-1], (4,1))+shift
        for i in range(4):
            x, y = top_left[i]
            xc, yc = center[i]
            color_center = False
            board_slice = board[x:x+half, y:y+half]
            if np.count_nonzero(board_slice) == slice_size:
                board[xc, yc] = 0
                color_center = True
            tile_board(board_slice)
            if color_center == True:
                board[xc, yc] = color
    return board

def make_board(n, x, y):
    size = 2**n
    board = np.ones((size, size))
    board[y, x] = 0
    tile_board(board)
    draw_board(board)

def make_interactive(n):
    size = 2**n
    return interactive(make_board, n=fixed(n), x=(0, size-1), y=(0, size-1))
